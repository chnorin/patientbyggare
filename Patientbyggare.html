Patientbyggare

Patient, akut fall och åtgärder byggs var för sig. Färg och prio räknas automatiskt ur RETTS. Severity styr både hur hårt och hur många parametrar som slår ut.

Normalspann = friska utgångsläget (slumpas vid start). Fysiologisk gräns = tak/golv som staplade lager aldrig passerar. Färg och prio bestäms av RETTS (se RETTS-fliken), inte av den här tabellen – parametrar med RETTS-koppling visas med RETTS.

create table if not exists parameters(key text primary key, label text, unit text, normal_min numeric, normal_max numeric, clamp_min numeric, clamp_max numeric, decimals int);
create table if not exists comorbidities(id text primary key, name text, distractor boolean, note text);
create table if not exists comorbidity_modifiers(comorbidity_id text references comorbidities(id) on delete cascade, param_key text, shift_min numeric, shift_max numeric);
create table if not exists patients(id text primary key, name text, age int, sex text, weight numeric, note text, o2_sensitive boolean, sat_target numeric);
create table if not exists patient_comorbidities(patient_id text references patients(id) on delete cascade, comorbidity_id text references comorbidities(id) on delete cascade);
create table if not exists conditions(id text primary key, name text, hidden_diagnosis text, dominant_system text, sev_min numeric, sev_max numeric, retts_floor text, correct_actions text, destination text, forvarning text, pitfalls text, source text, reviewer text, version text);
create table if not exists condition_overrides(condition_id text references conditions(id) on delete cascade, param_key text, abs_min numeric, abs_max numeric, sev_min numeric);
create table if not exists condition_findings(condition_id text references conditions(id) on delete cascade, abcde text, finding text, revealed_by text);
create table if not exists dispatches(id text primary key, name text, kod text, briefing text, note text);
create table if not exists dispatch_conditions(dispatch_id text references dispatches(id) on delete cascade, condition_id text references conditions(id) on delete cascade);
create table if not exists interventions(id text primary key, name text, route text, unit_label text, recommended_dose numeric, max_dose numeric, dose_step numeric, note text, source text, version text);
create table if not exists intervention_effects(intervention_id text references interventions(id) on delete cascade, kind text, param_key text, shift_min numeric, shift_max numeric, label text);
