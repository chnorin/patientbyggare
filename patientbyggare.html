create table if not exists condition_briefings(condition_id text references conditions(id) on delete cascade, template text);
-- slottar i template: {sedan} {tid} {namn} {ålder} {kön} {hanhon} {aktivitet}. Slumpa EN mall per scenario.
create table if not exists patient_pins(patient_id text references patients(id) on delete cascade, category text, value text);
-- pinnat värde styr över slumpen för den bakgrundskategorin.
create table if not exists condition_findings(condition_id text references conditions(id) on delete cascade, abcde text, finding text, revealed_by text);
create table if not exists dispatches(id text primary key, name text, kod text, briefing text, note text);
create table if not exists dispatch_conditions(dispatch_id text references dispatches(id) on delete cascade, condition_id text references conditions(id) on delete cascade);
create table if not exists interventions(id text primary key, name text, route text, unit_label text, recommended_dose numeric, max_dose numeric, dose_step numeric, note text, source text, version text);
create table if not exists intervention_effects(intervention_id text references interventions(id) on delete cascade, kind text, param_key text, shift_min numeric, shift_max numeric, label text, condition_ids text);
-- condition_ids: kommaseparerade fall-id. Tom = effekten verkar mot ALLA fall. Ifylld = bara mot dessa fall.
